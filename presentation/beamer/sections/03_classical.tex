      \bottomrule
    \end{tabularx}

  \vspace{0.35em}
  \begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
    \begin{panelbox}
      \textbf{Why straight lines appear}
      \begin{itemize}
        \item Rolling Mean is constant by design
        \item SES usually becomes a flat multi-step level forecast
        \item Holt extrapolates a straight trend
      \end{itemize}
    \end{panelbox}

    \column{0.48\textwidth}
    \begin{panelbox}
      \textbf{Why that is acceptable}
      When only 7 training points exist before the cutoff, simplicity is often the honest consequence of the setup rather than a weakness of the plot.
    \end{panelbox}
  \end{columns}

\begin{frame}[t]{Classical Lessons}
  \sectiontag{Classical}
  \takeaway{The classical notebook is useful because it exposes its limits clearly: train-window scarcity, not just model family choice, explains much of the observed behavior.}
  \vspace{0.4em}

  \begin{columns}[T,onlytextwidth]
    \column{0.31\textwidth}
    \begin{panelbox}
      \textbf{What classical models do well}
      \begin{itemize}
        \item smoothers remain stable when data is tiny
        \item local models are easy to interpret series-by-series
        \item ARIMA can exploit differencing on unstable series
      \end{itemize}
    \end{panelbox}

    \column{0.31\textwidth}
    \begin{panelbox}
      \textbf{What limits them here}
      \begin{itemize}
        \item two series have only seven training points
        \item many complex models are skipped by design
        \item fit feasibility becomes part of the result
      \end{itemize}
    \end{panelbox}

    \column{0.31\textwidth}
    \begin{panelbox}
      \textbf{Practical conclusion}
      For these notebooks, the classical story is not “complexity wins.” It is “complexity helps only when the data regime actually supports it.”
    \end{panelbox}
  \end{columns}
\end{frame}
