\begin{frame}[t]{Deep Lessons}
  \sectiontag{Deep}
  \takeaway{The deep notebook adds useful evidence, but it should be read as a sampled sequence-model study under adaptive splits, not as a definitive win over the classical notebook.}
  \vspace{0.4em}

  \begin{columns}[T,onlytextwidth]
    \column{0.31\textwidth}
    \begin{panelbox}
      \textbf{Where deep helped}
      \begin{itemize}
        \item highest-weight case improved meaningfully
        \item volatile case remained competitive
        \item recurrent models captured more structure than naive persistence on several panels
      \end{itemize}
    \end{panelbox}

    \column{0.31\textwidth}
    \begin{panelbox}
      \textbf{What did not change}
      \begin{itemize}
        \item the most stable series still behaves like a precision problem
        \item the models remain target-only
        \item the evidence is still limited to four representatives
      \end{itemize}
    \end{panelbox}

    \column{0.31\textwidth}
    \begin{panelbox}
      \textbf{Interpretation rule}
      The deep study is fair inside its own setup, but it cannot erase the split mismatch with the fixed-cutoff classical study.
    \end{panelbox}
  \end{columns}
\end{frame}
