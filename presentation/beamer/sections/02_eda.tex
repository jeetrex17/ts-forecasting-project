    \column{0.38\textwidth}
    \begin{softcard}
      {\bfseries Reading ADF and KPSS}\par
      \vspace{0.14em}
      \footnotesize
      \softbullet{\textbf{ADF} tests for unit-root style non-stationarity.}
      \softbullet{\textbf{KPSS} tests whether a stable level is plausible.}
      \softbullet{Using both together avoids one-test overconfidence.}
      \vspace{0.18em}
      \muted{\textbf{Modeling implication.}} Keep first differencing available later because the target is materially more stable after it.
    \end{softcard}
  \end{columns}
